SELECT 顧客ID,COUNT(*) 受注回数
FROM プロジェクト_W 
GROUP BY 顧客ID

5.
SELECT 社員コード,氏名, 回数
FROM 社員_W,(SELECT リーダー,COUNT(*) 回数
	FROM プロジェクト_W
	GROUP BY リーダー
	)
WHERE リーダー = 社員コード
AND   回数 >= 2;

FROM プロジェクト_W NATURAL JOIN スケジュール_W
WHERE 工程終了日-工程開始日 = (SELECT MAX(工程終了日-工程開始日)
				FROM スケジュール_W
				WHERE 工程コード = 'SI')
AND 工程コード = 'SI';
