01: #include <stdio.h>
02: #include <stdlib.h>
03: 
04: #define STACK_SIZE 5            /* 格納データの最大個数 */
05: 
06: int gStack[STACK_SIZE];         /* データ格納用の配列 */
07: int gSNum = 0;                  /* 格納データの個数 */
08: 
09: void initStack(void) {
10:   gSNum = 0;
11: }
12: 
13: int isStackEmpty(void) {
14:   return gSNum == 0;
15: }
16: 
17: int isStackFull(void) {
18:   return gSNum == STACK_SIZE;
19: }
20: 
21: void push(int x) {
22:   gStack[gSNum] = x;
23:   gSNum++;
24: }
25: 
26: int pop(void) {
27:   int x = gStack[gSNum - 1];
28:   gSNum--;
29:   return x;
30: }
31: 
32: void printStack(int tabs) {
33:   int i;
34: 
35:   for (i = 0; i < tabs; i++) printf("\t");
36:   printf("STACK[");
37:   for (i = 0; i < gSNum; i++) {
38:     printf("%d",  gStack[i]);
39:     if (i != gSNum - 1) printf(" ");
40:   }
41:   printf("]\n"); fflush(stdout);
42: }
43: 
44: int main(int argc, char *argv[]) {
45:   int i, x, y, sum=0;
46: 
47:   initStack();
48: 
49:   for (i = 1; i < argc; i++) {
50:     if (isStackFull()) break;
51:     x = atoi(argv[i]);
52:     if(x%2==0){
53:     printf("push(%d)", x);
54:     push(x);
55:     printStack(1);
56:     }
57:     }
58:   printf("----------\n");
59:   while (!isStackEmpty()) {
60:     y=pop();
61:     sum= sum+y;
62:     printf("pop() %d", y);
63:     printStack(1);
64:   }
65:   printf("----------\n");
66:   printf("合計 %d\n",sum);
67:   return 0;
68: }
