01: #include <stdio.h>
02: 
03: int main(void) {
04:   int    x = 77;
05:   double y = 3.14 ;
06: 
07:   /* (1) int  へのポインタ p を宣言します。*/
08:   int *p          ;
09: 
10:   /* (2) doubleへのポインタ q を宣言します。*/
11:   double  *q      ;
12: 
13:   /* (3) ポインタ p に変数 x のアドレスを代入します。*/
14:   p = &x          ;
15: 
16:   /* (4) ポインタ q に変数 y のアドレスを代入します。*/
17:   q = &y          ;
18: 
19:   printf("変数  アドレス          値               間接参照の値\n");
20:   printf("x     %p    %d       \n",     &x,     x);
21:   printf("p     %p    %p     %d\n",     &p,     p,      *p);
22: 
23:   /* (5) 変数 y とポインタ q だけを使って実行例のように出力します。*/
24:   printf("y     %p    %.2f     \n",&y       , y     );
25:   printf("q     %p    %p   %.2f\n",&q       , q     , *q       );
26: 
27:   /* (6) ポインタ p と q が指す変数の値の和を変数 x に代入します。*/
28:   x = *p + *q;
29: 
30:   printf("----------------------------------\n");
31:   printf("変数  アドレス          値               間接参照の値\n");
32:   printf("x     %p    %d       \n",     &x,     x);
33:   printf("p     %p    %p     %d\n",     &p,     p,      *p);
34: 
35:   /* (7) 変数 y とポインタ q だけを使って実行例のように出力します。(5)と同じ。*/
36:   printf("y     %p    %.2f     \n",     &y  ,    y  );
37:   printf("q     %p    %p   %.2f\n",     &q  ,     q ,      *q  );
38: 
39:   return 0;
40: }
