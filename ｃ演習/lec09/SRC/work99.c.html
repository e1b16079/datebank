#include<stdio.h>
#define N 8
int main(void){
   int a[N][N];
   int b[N][N];
   int i, j;

   for(i=0;i<N;i++){
     for(j=0;j<N;j++){
       a[i][j]=i*N+j;
       b[i][j]=i+j;
     }
   }
  
   for(i=0;i<N;i++){
     for(j=0;j<N;j++){
       printf("%3d",  a[i][j]+b[i][j]);
    }
    printf("\n");
   }
   return 0;
}
